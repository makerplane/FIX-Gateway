261 def _serial_send(self, command, ack=True, max_attempts=5): 

262 # Send a serial command and automatically handle if it needs to be resent 

263 # because of a bus error. If ack is True then an ackowledgement is 

264 # expected and only up to the maximum specified attempts will be made 

265 # to get a good acknowledgement (default is 5). If ack is False then 

266 # no acknowledgement is expected (like when resetting the device). 

267 attempts = 0 

268 while True: 

269 # Flush any pending received data to get into a clean state. 

270 self._serial.flushInput() 

271 # Send the data. 

272 self._serial.write(command) 

273 logger.debug("Serial send: 0x{0}".format(binascii.hexlify(command))) 

274 # Stop if no acknowledgment is expected. 

275 if not ack: 

276 return 

277 # Read acknowledgement response (2 bytes). 

278 resp = bytearray(self._serial.read(2)) 

279 logger.debug("Serial receive: 0x{0}".format(binascii.hexlify(resp))) 

280 if resp is None or len(resp) != 2: 

281 raise RuntimeError( 

282 "Timeout waiting for serial acknowledge, is the BNO055 connected?" 

283 ) 

284 # Stop if there's no bus error (0xEE07 response) and return response bytes. 

285 if not (resp[0] == 0xEE and resp[1] == 0x07): 

286 return resp 

287 # Else there was a bus error so resend, as recommended in UART app 

288 # note at: 

289 # http://ae-bst.resource.bosch.com/media/products/dokumente/bno055/BST-BNO055-AN012-00.pdf 

290 attempts += 1 

291 if attempts >= max_attempts: 

292 raise RuntimeError( 

293 "Exceeded maximum attempts to acknowledge serial command without bus error!" 

294 )

296 def _write_bytes(self, address, data, ack=True): 

297 # Write a list of 8-bit values starting at the provided register address. 

298 if self._i2c_device is not None: 

299 # I2C write. 

300 self._i2c_device.writeList(address, data) 

301 else: 

302 # Build and send serial register write command. 

303 command = bytearray(4 + len(data)) 

304 command[0] = 0xAA # Start byte 

305 command[1] = 0x00 # Write 

306 command[2] = address & 0xFF 

307 command[3] = len(data) & 0xFF 

308 command[4:] = map(lambda x: x & 0xFF, data) 

309 resp = self._serial_send(command, ack=ack) 

310 # Verify register write succeeded if there was an acknowledgement. 

311 if resp[0] != 0xEE and resp[1] != 0x01: 

312 raise RuntimeError( 

313 "Register write error: 0x{0}".format(binascii.hexlify(resp)) 

314 ) 

315 

316 def _write_byte(self, address, value, ack=True): 

317 # Write an 8-bit value to the provided register address. If ack is True 

318 # then expect an acknowledgement in serial mode, otherwise ignore any 

319 # acknowledgement (necessary when resetting the device). 

320 if self._i2c_device is not None: 

321 # I2C write. 

322 self._i2c_device.write8(address, value) 

323 else: 

324 # Build and send serial register write command. 

325 command = bytearray(5) 

326 command[0] = 0xAA # Start byte 

327 command[1] = 0x00 # Write 

328 command[2] = address & 0xFF 

329 command[3] = 1 # Length (1 byte) 

330 command[4] = value & 0xFF 

331 resp = self._serial_send(command, ack=ack) 

332 # Verify register write succeeded if there was an acknowledgement. 

333 if ack and resp[0] != 0xEE and resp[1] != 0x01: 

334 raise RuntimeError( 

335 "Register write error: 0x{0}".format(binascii.hexlify(resp)) 

336 )

338 def _read_bytes(self, address, length): 

339 # Read a number of unsigned byte values starting from the provided address. 

340 if self._i2c_device is not None: 

341 # I2C read. 

342 return bytearray(self._i2c_device.readList(address, length)) 

343 else: 

344 # Build and send serial register read command. 

345 command = bytearray(4) 

346 command[0] = 0xAA # Start byte 

347 command[1] = 0x01 # Read 

348 command[2] = address & 0xFF 

349 command[3] = length & 0xFF 

350 resp = self._serial_send(command) 

351 # Verify register read succeeded. 

352 if resp[0] != 0xBB: 

353 raise RuntimeError( 

354 "Register read error: 0x{0}".format(binascii.hexlify(resp)) 

355 ) 

356 # Read the returned bytes. 

357 length = resp[1] 

358 resp = bytearray(self._serial.read(length)) 

359 logger.debug("Received: 0x{0}".format(binascii.hexlify(resp))) 

360 if resp is None or len(resp) != length: 

361 raise RuntimeError( 

362 "Timeout waiting to read data, is the BNO055 connected?" 

363 ) 

364 return resp

366 def _read_byte(self, address): 

367 # Read an 8-bit unsigned value from the provided register address. 

368 if self._i2c_device is not None: 

369 # I2C read. 

370 return self._i2c_device.readU8(address) 

371 else: 

372 return self._read_bytes(address, 1)[0]

374 def _read_signed_byte(self, address): 

375 # Read an 8-bit signed value from the provided register address. 

376 data = self._read_byte(address) 

377 if data > 127: 

378 return data - 256 

379 else: 

380 return data

390 def begin(self, mode=OPERATION_MODE_NDOF): 

391 """Initialize the BNO055 sensor. Must be called once before any other 

392 BNO055 library functions. Will return True if the BNO055 was 

393 successfully initialized, and False otherwise. 

394 """ 

395 # Save the desired normal operation mode. 

396 self._mode = mode 

397 # First send a thow-away command and ignore any response or I2C errors 

398 # just to make sure the BNO is in a good state and ready to accept 

399 # commands (this seems to be necessary after a hard power down). 

400 try: 

401 self._write_byte(BNO055_PAGE_ID_ADDR, 0, ack=False) 

402 except IOError: 

403 # Swallow an IOError that might be raised by an I2C issue. Only do 

404 # this for this very first command to help get the BNO and board's 

405 # I2C into a clear state ready to accept the next commands. 

406 pass 

407 # Make sure we're in config mode and on page 0. 

408 self._config_mode() 

409 self._write_byte(BNO055_PAGE_ID_ADDR, 0) 

410 # Check the chip ID 

411 bno_id = self._read_byte(BNO055_CHIP_ID_ADDR) 

412 logger.debug("Read chip ID: 0x{0:02X}".format(bno_id)) 

413 if bno_id != BNO055_ID: 

414 return False 

415 # Reset the device. 

416 if self._rst is not None: 

417 # Use the hardware reset pin if provided. 

418 # Go low for a short period, then high to signal a reset. 

419 self._gpio.set_low(self._rst) 

420 time.sleep(0.01) # 10ms 

421 self._gpio.set_high(self._rst) 

422 else: 

423 # Else use the reset command. Note that ack=False is sent because 

424 # the chip doesn't seem to ack a reset in serial mode (by design?). 

425 self._write_byte(BNO055_SYS_TRIGGER_ADDR, 0x20, ack=False) 

426 # Wait 650ms after reset for chip to be ready (as suggested 

427 # in datasheet). 

428 time.sleep(0.65) 

429 # Set to normal power mode. 

430 self._write_byte(BNO055_PWR_MODE_ADDR, POWER_MODE_NORMAL) 

431 # Default to internal oscillator. 

432 self._write_byte(BNO055_SYS_TRIGGER_ADDR, 0x0) 

433 # Enter normal operation mode. 

434 self._operation_mode() 

435 return True

437 def set_mode(self, mode): 

438 """Set operation mode for BNO055 sensor. Mode should be a value from 

439 table 3-3 and 3-5 of the datasheet: 

440 http://www.adafruit.com/datasheets/BST_BNO055_DS000_12.pdf 

441 """ 

442 self._write_byte(BNO055_OPR_MODE_ADDR, mode & 0xFF) 

443 # Delay for 30 milliseconds (datsheet recommends 19ms, but a little more 

444 # can't hurt and the kernel is going to spend some unknown amount of time 

445 # too). 

446 time.sleep(0.03)

448 def get_revision(self): 

449 """Return a tuple with revision information about the BNO055 chip. Will 

450 return 5 values: 

451 - Software revision 

452 - Bootloader version 

453 - Accelerometer ID 

454 - Magnetometer ID 

455 - Gyro ID 

456 """ 

457 # Read revision values. 

458 accel = self._read_byte(BNO055_ACCEL_REV_ID_ADDR) 

459 mag = self._read_byte(BNO055_MAG_REV_ID_ADDR) 

460 gyro = self._read_byte(BNO055_GYRO_REV_ID_ADDR) 

461 bl = self._read_byte(BNO055_BL_REV_ID_ADDR) 

462 sw_lsb = self._read_byte(BNO055_SW_REV_ID_LSB_ADDR) 

463 sw_msb = self._read_byte(BNO055_SW_REV_ID_MSB_ADDR) 

464 sw = ((sw_msb << 8) | sw_lsb) & 0xFFFF 

465 # Return the results as a tuple of all 5 values. 

466 return (sw, bl, accel, mag, gyro)

468 def set_external_crystal(self, external_crystal): 

469 """Set if an external crystal is being used by passing True, otherwise 

470 use the internal oscillator by passing False (the default behavior). 

471 """ 

472 # Switch to configuration mode. 

473 self._config_mode() 

474 # Set the clock bit appropriately in the SYS_TRIGGER register. 

475 if external_crystal: 

476 self._write_byte(BNO055_SYS_TRIGGER_ADDR, 0x80) 

477 else: 

478 self._write_byte(BNO055_SYS_TRIGGER_ADDR, 0x00) 

479 # Go back to normal operation mode. 

480 self._operation_mode()

482 def get_system_status(self, run_self_test=True): 

483 """Return a tuple with status information. Three values will be returned: 

484 - System status register value with the following meaning: 

485 0 = Idle 

486 1 = System Error 

487 2 = Initializing Peripherals 

488 3 = System Initialization 

489 4 = Executing Self-Test 

490 5 = Sensor fusion algorithm running 

491 6 = System running without fusion algorithms 

492 - Self test result register value with the following meaning: 

493 Bit value: 1 = test passed, 0 = test failed 

494 Bit 0 = Accelerometer self test 

495 Bit 1 = Magnetometer self test 

496 Bit 2 = Gyroscope self test 

497 Bit 3 = MCU self test 

498 Value of 0x0F = all good! 

499 - System error register value with the following meaning: 

500 0 = No error 

501 1 = Peripheral initialization error 

502 2 = System initialization error 

503 3 = Self test result failed 

504 4 = Register map value out of range 

505 5 = Register map address out of range 

506 6 = Register map write error 

507 7 = BNO low power mode not available for selected operation mode 

508 8 = Accelerometer power mode not available 

509 9 = Fusion algorithm configuration error 

510 10 = Sensor configuration error 

511 

512 If run_self_test is passed in as False then no self test is performed and 

513 None will be returned for the self test result. Note that running a 

514 self test requires going into config mode which will stop the fusion 

515 engine from running. 

516 """ 

517 self_test = None 

518 if run_self_test: 

519 # Switch to configuration mode if running self test. 

520 self._config_mode() 

521 # Perform a self test. 

522 sys_trigger = self._read_byte(BNO055_SYS_TRIGGER_ADDR) 

523 self._write_byte(BNO055_SYS_TRIGGER_ADDR, sys_trigger | 0x1) 

524 # Wait for self test to finish. 

525 time.sleep(1.0) 

526 # Read test result. 

527 self_test = self._read_byte(BNO055_SELFTEST_RESULT_ADDR) 

528 # Go back to operation mode. 

529 self._operation_mode() 

530 # Now read status and error registers. 

531 status = self._read_byte(BNO055_SYS_STAT_ADDR) 

532 error = self._read_byte(BNO055_SYS_ERR_ADDR) 

533 # Return the results as a tuple of all 3 values. 

534 return (status, self_test, error)

536 def get_calibration_status(self): 

537 """Read the calibration status of the sensors and return a 4 tuple with 

538 calibration status as follows: 

539 - System, 3=fully calibrated, 0=not calibrated 

540 - Gyroscope, 3=fully calibrated, 0=not calibrated 

541 - Accelerometer, 3=fully calibrated, 0=not calibrated 

542 - Magnetometer, 3=fully calibrated, 0=not calibrated 

543 """ 

544 # Return the calibration status register value. 

545 cal_status = self._read_byte(BNO055_CALIB_STAT_ADDR) 

546 sys = (cal_status >> 6) & 0x03 

547 gyro = (cal_status >> 4) & 0x03 

548 accel = (cal_status >> 2) & 0x03 

549 mag = cal_status & 0x03 

550 # Return the results as a tuple of all 3 values. 

551 return (sys, gyro, accel, mag)

584 def get_axis_remap(self): 

585 """Return a tuple with the axis remap register values. This will return 

586 6 values with the following meaning: 

587 - X axis remap (a value of AXIS_REMAP_X, AXIS_REMAP_Y, or AXIS_REMAP_Z. 

588 which indicates that the physical X axis of the chip 

589 is remapped to a different axis) 

590 - Y axis remap (see above) 

591 - Z axis remap (see above) 

592 - X axis sign (a value of AXIS_REMAP_POSITIVE or AXIS_REMAP_NEGATIVE 

593 which indicates if the X axis values should be positive/ 

594 normal or negative/inverted. The default is positive.) 

595 - Y axis sign (see above) 

596 - Z axis sign (see above) 

597 

598 Note that by default the axis orientation of the BNO chip looks like 

599 the following (taken from section 3.4, page 24 of the datasheet). Notice 

600 the dot in the corner that corresponds to the dot on the BNO chip: 

601 

602 | Z axis 

603 | 

604 | / X axis 

605 ____|__/____ 

606 Y axis / * | / /| 

607 _________ /______|/ // 

608 /___________ // 

609 |____________|/ 

610 """ 

611 # Get the axis remap register value. 

612 map_config = self._read_byte(BNO055_AXIS_MAP_CONFIG_ADDR) 

613 z = (map_config >> 4) & 0x03 

614 y = (map_config >> 2) & 0x03 

615 x = map_config & 0x03 

616 # Get the axis remap sign register value. 

617 sign_config = self._read_byte(BNO055_AXIS_MAP_SIGN_ADDR) 

618 x_sign = (sign_config >> 2) & 0x01 

619 y_sign = (sign_config >> 1) & 0x01 

620 z_sign = sign_config & 0x01 

621 # Return the results as a tuple of all 3 values. 

622 return (x, y, z, x_sign, y_sign, z_sign)

624 def set_axis_remap( 

625 self, 

626 x, 

627 y, 

628 z, 

629 x_sign=AXIS_REMAP_POSITIVE, 

630 y_sign=AXIS_REMAP_POSITIVE, 

631 z_sign=AXIS_REMAP_POSITIVE, 

632 ): 

633 """Set axis remap for each axis. The x, y, z parameter values should 

634 be set to one of AXIS_REMAP_X, AXIS_REMAP_Y, or AXIS_REMAP_Z and will 

635 change the BNO's axis to represent another axis. Note that two axises 

636 cannot be mapped to the same axis, so the x, y, z params should be a 

637 unique combination of AXIS_REMAP_X, AXIS_REMAP_Y, AXIS_REMAP_Z values. 

638 

639 The x_sign, y_sign, z_sign values represent if the axis should be positive 

640 or negative (inverted). 

641 

642 See the get_axis_remap documentation for information on the orientation 

643 of the axises on the chip, and consult section 3.4 of the datasheet. 

644 """ 

645 # Switch to configuration mode. 

646 self._config_mode() 

647 # Set the axis remap register value. 

648 map_config = 0x00 

649 map_config |= (z & 0x03) << 4 

650 map_config |= (y & 0x03) << 2 

651 map_config |= x & 0x03 

652 self._write_byte(BNO055_AXIS_MAP_CONFIG_ADDR, map_config) 

653 # Set the axis remap sign register value. 

654 sign_config = 0x00 

655 sign_config |= (x_sign & 0x01) << 2 

656 sign_config |= (y_sign & 0x01) << 1 

657 sign_config |= z_sign & 0x01 

658 self._write_byte(BNO055_AXIS_MAP_SIGN_ADDR, sign_config) 

659 # Go back to normal operation mode. 

660 self._operation_mode()

662 def _read_vector(self, address, count=3): 

663 # Read count number of 16-bit signed values starting from the provided 

664 # address. Returns a tuple of the values that were read. 

665 data = self._read_bytes(address, count * 2) 

666 result = [0] * count 

667 for i in range(count): 

668 result[i] = ((data[i * 2 + 1] << 8) | data[i * 2]) & 0xFFFF 

669 if result[i] > 32767: 

670 result[i] -= 65536 

671 return result
